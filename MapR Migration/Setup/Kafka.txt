sudo bash
cd $KAFKA_HOME
kafka-server-start.sh config/server.properties > logs/kafka-server.log 2>&1 &
exit

kafka-topics.sh --create \
  --zookeeper onprem:5181 \
  --replication-factor 1 \
  --partitions 1 \
  --topic flights

  kafka-topics.sh --list \
  --zookeeper onprem:5181

cd ~/apps/kafka

java -cp EventProducer.jar eventdriver \
      --bootstrap-servers onprem:9092 \
      --sourcefilename 2000.csv \
      --topic flights \
      --partitions 1 > /dev/null &

java -cp Consumer.jar eventconsumer \
      --bootstrap-servers onprem:9092 \
      --topic flights \
      --partitions 1
